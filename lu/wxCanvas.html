canvas绘制分享海报

无问西东 2019/11/16 21:02:00 微信小程序 canvas
  1. 首先，需要写一个canvas标签，给canvas-id命名为Poster。
<canvas class="canvas" canvas-id="Poster"></canvas>
  1. 为了在页面隐藏canvas，我们用绝对定位把它移出视口。
.canvas{
  position: fixed;
  top: 0;
  left: 1000rpx;
  width: 640px;
  height: 1280px;
}

3.开始绘制

# 3.1 绘制背景图

  • 图片只能是远程图片，通过wx.getImageInfo取得本地路径

  • ctx.draw()将之前在绘图上下文中的描述（路径、变形、样式）画到 canvas 中。

    第一个参数表示本次绘制是否接着上一次绘制，false会清空画布再绘制，默认false。

    第二个参数是绘制完成后执行的回调函数。

createCanvas: function () {
    wx.showLoading({
      title: '正在生成海报',
      mask: true
    })
    let that = this;
    let ctx = wx.createCanvasContext('Poster');
    let bgImg = 'https://www.site.com/images/bg.png';
    ctx.save();
    wx.getImageInfo({
      src: bgImg,
      success: function (res) {
        ctx.drawImage(res.path, 0, 0, 640, 1280);
        ctx.restore();
        ctx.draw(true, function () {
          that.drawLogo(ctx);
        })
      },
      fail: function (res) {
        wx.hideLoading();
        wx.showToast({
          title: '生成海报失败，请重试.(bg_fail)',
          icon: 'none'
        })
      },
    });
  },

# 3.2 绘制logo

drawLogo: function (ctx) {
    let that = this;
    let logo = 'https://www.site.com/images/logo.png';
    // 画下圆角的白色矩形
    ctx.save();
    ctx.setFillStyle('#fff');
    ctx.beginPath();
    ctx.moveTo(218, 0);
    ctx.lineTo(422, 0);
    ctx.lineTo(422, 46);
    ctx.arc(414, 46, 8, 0, 0.5 * Math.PI)
    ctx.lineTo(224, 54);
    ctx.arc(224, 46, 8, 0.5 * Math.PI, 1 * Math.PI);
    ctx.closePath();
    ctx.fill();
    ctx.restore();
    wx.getImageInfo({
      src: logo,
      success: function (res) {
        ctx.save();
        ctx.beginPath()
        ctx.moveTo(218, 0);
        ctx.lineTo(422, 0);
        ctx.lineTo(422, 54);
        ctx.lineTo(218, 54);
        ctx.closePath();
        ctx.clip()
        ctx.drawImage(res.path, 250, -46, 140, 140);
        ctx.restore();
        ctx.draw(true, function () {
          that.drawQrCode(ctx);
        });
      },
      fail: function (res) {
        wx.hideLoading();
        wx.showToast({
          title: '生成海报失败，请重试.(logo_fail)',
          icon: 'none'
        })
      },
    });
  },

# 3.6 绘制圆角矩形

  • 参数 r 是圆角半径
  • 参数 rect 是包含矩形的坐标和长宽的object
  • 参数 color 是矩形的背景色
  • 圆角的角度说明参见下图
drawRoundedRect: function (ctx, r, rect, color) {
  ctx.save();
  ctx.setFillStyle(color);
  ctx.beginPath();
  ctx.moveTo(rect.x + r, rect.y);
  ctx.lineTo(rect.x + rect.width - r, rect.y);
  ctx.arc(rect.x + rect.width - r, rect.y + r, r, Math.PI * 1.5, 0, false);
  ctx.lineTo(rect.x + rect.width, rect.y + rect.height - r);
  ctx.arc(rect.x + rect.width - r, rect.y + rect.height - r, r, 0, Math.PI * 0.5, 0, false);
  ctx.lineTo(rect.x + r, rect.y + rect.height);
  ctx.arc(rect.x + r, rect.y + rect.height - r, r, Math.PI * 0.5, Math.PI, false);
  ctx.lineTo(rect.x, rect.y + r);
  ctx.arc(rect.x + r, rect.y + r, r, Math.PI, Math.PI * 1.5, false);
  ctx.closePath();
  ctx.fill();
  ctx.restore();
},

# 3.7 绘制多行文本

  • @str 文本内容
  • @lineWidth 行宽
  • @lineHeight 行高
  • @x 起始点横坐标
  • @y 起始点纵坐标
  • 逐字测量文本宽度strWidth，当超过行宽时记录下标值（用以后续换行），重新记录文本宽度strWidth。
drawTextarea: function (ctx, str, lineWidth, lineHeight, fontSize, fontColor, x, y) {
  ctx.save();
  let strWidth = 0;
  let lineNum = 1;
  let splitArr = [0]
  for (var i = 0; i < str.length; i++) {
    strWidth += ctx.measureText(str[i]).width;
    if (strWidth > lineWidth) {
      splitArr.push(i - 1);
      strWidth = ctx.measureText(str[i]).width;
    }
  }
  for (var j = 0; j < splitArr.length; j++) {
    let stext = '';
    if (splitArr[j + 1]) {
      stext = str.substring(splitArr[j], splitArr[j + 1])
    } else {
      stext = str.substring(splitArr[j])
    }
    // console.log(stext);
    ctx.setFontSize(fontSize);
    ctx.setFillStyle(fontColor);
    ctx.fillText(stext, x, y + j * lineHeight);
  }
  ctx.restore();
},

# 3.8 把canvas转存为图片

canvasToTempFilePath: function () {
  let that = this;
  wx.canvasToTempFilePath({
    x: 0,
    y: 0,
    width: 640,
    height: 1280,
    destWidth: 1280,
    destHeight: 2560,
    canvasId: 'tuPaiPoster',
    success(res) {
      wx.hideLoading();
      wx.saveImageToPhotosAlbum({
        filePath: res.tempFilePath,
        success: function () {
          that.setData({
            tempImg: res.tempFilePath
          })
        },
        fail: function (err) {
        }
      })
    }
  })
}
